	{() => {
		const knobs = {
				literals: object('literals', [{
					title: 'Lorem ipsum dolor sit amet',
					body: 'Sed ut perspiciatis unde omnis iste natus error sit voluptatem accusantium doloremque laudantium.'
				}, {
					title: 'Lorem ipsum dolor sit amet',
					body: 'Sed ut perspiciatis unde omnis iste natus error sit voluptatem accusantium doloremque laudantium.'
				}, {
					title: 'Lorem ipsum dolor sit amet',
					body: 'Sed ut perspiciatis unde omnis iste natus error sit voluptatem accusantium doloremque laudantium.'
				}, {
					title: 'Lorem ipsum dolor sit amet',
					body: 'Sed ut perspiciatis unde omnis iste natus error sit voluptatem accusantium doloremque laudantium.'
				}
				]),
				title: text('[SLOT] title', 'Lorem ipsum dolor sit amet consectetur adipiscing elit'),
				textButton: text('[SLOT] text Button', 'Lorem ipsum dolor')
		}
		return html`
			<correos-cdk-text-block literals='${JSON.stringify(knobs.literals)}'>
				<h3 slot="title">${knobs.title}</h3>
				<correos-ui-button
					slot="link"
					theme="secondary"
					variant="link"
					link="true"
					@click='${action('button-click')}'
				>
					<span slot="text">${knobs.textButton}</span>
					<i slot="icon-right" class="icon-download"></i>
				</correos-ui-button>
			</correos-cdk-text-block>
		`;
	}}
</Story>
