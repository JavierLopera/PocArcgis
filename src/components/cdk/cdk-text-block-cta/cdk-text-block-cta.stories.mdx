  {() => {
	  const knobs = {
				header: text('[SLOT] header', 'Lorem ipsum dolor sit amet consectetur adipiscing elit'),
				title: text('[SLOT] title', '¿Cunto tiempo hace que no charlas con el portero?'),

		}
		return html`
			<correos-cdk-text-block-cta>
				<p slot="header">${knobs.header}</p>
				<h3 slot="title">${knobs.title}</h3>
				<p slot="text">${knobs.text}</p>
				<correos-ui-button
					slot="link"
					theme="secondary"
					variant="link"
					link="true"
					@click='${action('button-click')}'
				>
					<span slot="text">${knobs.textButton}</span>
					<i slot="icon-right" class="icon-download"></i>
				</correos-ui-button>
			</correos-cdk-text-block-cta>
		`}
