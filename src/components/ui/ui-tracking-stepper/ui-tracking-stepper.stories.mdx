## Story definition

Preview of default state.

<Story name="Develop">
  {() => {
		const points = [
			{ done: true, icon: 'icon-notif_ok', type: 'secondary', title: 'Preadmisión', date: '27 de Mayo, 2019 · 10:30', childs: [
                {
                    done: true, icon: false, type: 'default', title: 'Asignación correcta', date: '27 de Mayo, 2019 · 10:31', desc: 'Identificado al remitente'
                }
            ] },
			{ done: true, icon: 'icon-notif_ok', type: 'secondary', title: 'En camino', date: '28 de Mayo, 2019 · 09:00' },
			{ done: true, icon: 'icon-entrega', type: 'primary', title: 'En entrega', desc: "El envío está en reparto", date: '28 de Mayo, 2019 · 14:31', childs: [
                {
                    done: true, icon: false, type: 'default', title: 'Alta en unidad de reparto', date: '28 de Mayo, 2019 · 14:35', desc: 'Envio en reparto'
                },
                {
                    done: false, icon: false, type: 'default', title: 'En reparto', desc: 'Pendiente de salida a reparto'
                },
            ] },
			{ done: false, icon: false, type: 'default', title: 'Entregado' }
        ];
		const knobs = {
			size: optionsKnob('Tamaño',
					{
						'Computer': 'lg',
						'Tablet': 'md',
						'Phone': 'sm'
					},
					'sm',
					{ display: 'select' }
			),
			type: optionsKnob('Tipo',
					{
						'Horizontal': 'horizontal',
						'Vertical': 'vertical'
					},
					'horizontal',
					{ display: 'select' }
			),
			points: text('Puntos', JSON.stringify(points))
		};
		return html`
			<correos-ui-tracking-stepper
				type='${knobs.type}'
				points='${knobs.points}'
			/>
		`}
	}
</Story>
