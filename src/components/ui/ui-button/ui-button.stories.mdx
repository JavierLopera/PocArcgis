## Story definition

Preview of default state button.

<Story name="Develop button">
	{() => {
		const knobs = {
			attr: {
				theme: optionsKnob('Theme',
					{
						'Default': '',
						'Primary': 'primary',
						'Secondary': 'secondary',
						'Ghost': 'ghost'
					},
					'primary',
					{ display: 'select' },
					'Button'
				),
				disabled: boolean('Disabled', false,'Button'),
				type: optionsKnob('Type',
					{
						'Text': 'text',
						'Submit': 'submit'
					},
					'text',
					{ display: 'select' },
					'Button'
				)
			},
			slot: {
				text: text('Text', 'text Button', 'Slot'),
				iconLeftClass: optionsKnob('Left icon',
					{
						'No icon': 'display-none',
						'icon-download': 'icon-download',
						'icon-arrow': 'icon-arrow',
						'icon-notif_info': 'icon-notif_info'
					},
					'icon-download',
					{ display: 'select' },
					'Slot'
				)
			}
		};
		return html`
			<correos-ui-button
				theme='${knobs.attr.theme}'
				disabled='${knobs.attr.disabled}'
				type='${knobs.attr.type}'
				@click='${action('button-click')}'
			>
				<i slot="icon-left" class="${knobs.slot.iconLeftClass}" ></i>
				<span slot="text">${knobs.slot.text}</span>
			</correos-ui-button>
		`;
	}}
</Story>

<Story name="Develop link">
	{() => {
		const knobs = {
			attr: {
				theme: optionsKnob('Theme',
					{
						'Primary': 'primary',
						'Secondary': 'secondary',
					},
					'secondary',
					{ display: 'select' },
					'Link'
				),
				disabled: boolean('Disabled', false, 'Link')
			},
			slot: {
				text: text('Text', 'text Button', 'Slot'),
				iconRightClass: optionsKnob('Right icon',
					{
						'No icon': 'display-none',
						'icon-download': 'icon-download',
						'icon-arrow': 'icon-arrow',
						'icon-notif_info': 'icon-notif_info'
					},
					'icon-download',
					{ display: 'select' },
					'Slot'
				)
			}
		};
		return html`
			<correos-ui-button
				theme='${knobs.attr.theme}'
				variant='link'
				disabled='${knobs.attr.disabled}'
				hrefLink="/ejemplo"
				@click='${action('button-click')}'
			>
				<i slot="icon-right" class="${knobs.slot.iconRightClass}" ></i>
				<span slot="text">${knobs.slot.text}</span>
			</correos-ui-button>
		`;
	}}
</Story>
