<Story name="Develop direct access button">
	{() => {
		const slot = {
			text: text("[SLOT] text", "Quiero hacer una devolución"),
			iconClassName: optionsKnob(
				"[SLOT] icon",
				{
					Default: "display-none",
					"icon-almacen": "icon-almacen",
					"icon-download": "icon-download"
				},
				"icon-almacen",
				{ display: "select" }
			),
			disabled: boolean("disabled", false)
		};
		{
			/* ${slotCardTitle.disabled ? 'disabled aria-disabled="true"' : 'aria-disabled="false"'} */
		}
		return html`
			<correos-ui-card style="width:192px" disabled=${slot.disabled}>
				<i slot="icon" class="${slot.iconClassName}"></i>
				<p slot="text">${slot.text}</p>
			</correos-ui-card>
		`;
	}}
</Story>
